\section*{Solutions}

\begin{solution}{1}
    The capital of France is Paris.
\end{solution}

\begin{solution}{2}
    To solve the equation \( x^2 - 5x + 6 = 0 \), we factor it as \( (x - 2)(x - 3) = 0 \), so the solutions are \( x = 2 \) and \( x = 3 \).
\end{solution}

\begin{solution}{3}
    We know that \( \int \sin(x) \, dx = -\cos(x) + C \). Evaluating from \(0\) to \(\pi\):
    \[ \int_{0}^{\pi} \sin(x) \, dx = -\cos(\pi) - (-\cos(0)) = -(-1) - (-1) = 2 \]
\end{solution}
